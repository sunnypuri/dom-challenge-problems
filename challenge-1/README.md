## Template

> HTML
```
<body>
    <div id="star"></div>
    <div id="display-star"></div>


    <script src="script.js"></script>
    <script>
        function getStar(value){
            document.getElementById("display-star").innerHTML = value;

        new Star("#star", 5, getStar);
```
